CREATE TABLE utilizador (
	user_name VARCHAR(512),
	email	 VARCHAR(512),
	password	 VARCHAR(512),
	estado	 BOOL,
	avaliacao INTEGER,
	admin	 BOOL,
	PRIMARY KEY(user_name,email)
);

CREATE TABLE leilao (
	titulo		 VARCHAR(512),
	ean_artigo		 INTEGER,
	estado		 INTEGER,
	data_final		 TIMESTAMP,
	data_inicial	 TIMESTAMP,
	preco_min		 FLOAT(8),
	utilizador_user_name VARCHAR(512) NOT NULL,
	utilizador_email	 VARCHAR(512) NOT NULL,
	PRIMARY KEY(ean_artigo)
);

CREATE TABLE licitacao (
	valor		 FLOAT(8),
	data		 TIMESTAMP,
	validacao		 BOOL,
	utilizador_user_name VARCHAR(512) NOT NULL,
	utilizador_email	 VARCHAR(512) NOT NULL,
	leilao_ean_artigo	 INTEGER NOT NULL
);

CREATE TABLE mensagem (
	descricao		 VARCHAR(512),
	data		 TIMESTAMP,
	leilao_ean_artigo	 INTEGER NOT NULL,
	utilizador_user_name VARCHAR(512) NOT NULL,
	utilizador_email	 VARCHAR(512) NOT NULL
);

CREATE TABLE descricao (
	descricao	 VARCHAR(512),
	data		 TIMESTAMP,
	leilao_ean_artigo INTEGER NOT NULL
);

CREATE TABLE notificacao (
	estado		 BOOL DEFAULT False,
	descricao		 VARCHAR(512),
	leilao_ean_artigo	 INTEGER NOT NULL,
	utilizador_user_name VARCHAR(512) NOT NULL,
	utilizador_email	 VARCHAR(512) NOT NULL
);

CREATE TABLE token (
	valor		 BIGINT,
	validade		 TIMESTAMP,
	utilizador_user_name VARCHAR(512) NOT NULL,
	utilizador_email	 VARCHAR(512) NOT NULL,
	PRIMARY KEY(valor)
);
